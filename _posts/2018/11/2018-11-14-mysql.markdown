        </strong>
    </span>
</p>
```bash
sudo mysql -u root -p
#然后使用以下命令更新密码(其中new-password为前面设置的密码)
ALTER USER 'root'@'localhost' IDENTIFIED WITH mysql_native_password BY 'new-password';

#一旦完成stop and start了mysql服务器。
sudo service mysql stop
sudo service mysql start
```
